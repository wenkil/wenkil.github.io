print('百度key', os.getenv("QIANFAN_ACCESS_KEY"))
print('百度key', os.getenv("QIANFAN_SECRET_KEY"))


def chat_with_qianfan(prompt, input):
    print(f"百度方法接收参数：\nPrompt: {prompt}\nInput: {input}")

    # 初始化 ChatCompletion 对象
    chat_comp = qianfan.ChatCompletion()

    # 调用模型并获取响应
    # 可用模型：ERNIE-Speed-128k ERNIE-Speed-8K
    try:
        resp = chat_comp.do(
            model='ERNIE-Speed-128k',
            messages=[{
                "role": "user",
                "content": input
            }],
            system=prompt
        )
        print('结果返回', resp["body"]["result"])
        return resp["body"]["result"]
    except Exception as e:
        print(f"An error occurred: {e}")
        return None

# 运行当前脚本测试 运行crew.py时需要注掉此行
chat_with_qianfan( '智能助手', '写一首诗')

  3. 自定义工具tool.py

    import os
import requests
from langchain.tools import BaseTool
from pydantic import Field
from qianfan_api import chat_with_qianfan

class CustomAITool(BaseTool):
    """自定义AI工具，用于查询OpenAI API"""

    name: str = "CustomAITool"
    description: str = "使用给定提示查询模型API的工具。"
    api_key: str = Field(default=None)
    api_url: str = Field(default=None)
    api_secret: str = Field(default=None)
    model: str = ""

    # 根据模型名称设置API URL
    def set_api_url_based_on_model(self):
        self.model = os.getenv("AI_MODEL")  # 获取环境变量，默认为"default_model"
        print('当前环境model', self.model)
        if self.model == "default_model":
            self.api_key = os.getenv("OPENAI_API_KEY")
            self.api_url = os.getenv("OPENAI_BASE_URL")

    def __init__(self):
        super().__init__()
        self.set_api_url_based_on_model()  # 根据模型设置API URL
        print('CustomAITool初始化完成---api_url:', self.api_url)
        print('CustomAITool初始化完成---api_key:', self.api_key)
        print('CustomAITool初始化完成---api_secret:', self.api_secret)

    def _run(self, tool_input: str, *args) -> str:
        """使用给定提示查询OpenAI API

        参数:
            tool_input (str): 发送到API的提示。

        返回:
            str: 来自API的响应。
        """
        headers = {
            'Authorization': f'Bearer {self.api_key}',
            'Content-Type': 'application/json'
        }
        data = {
            'prompt': tool_input
        }
        print('CustomAI发送请求---api_url:', self.api_url)
        print('CustomAI发送请求---data:', data)
        print('CustomAI发送请求---headers:', headers)
        print('其他参数:', args)

        if self.model == "qianfan":
            try:
                # concatenate_if_tuple = lambda t_args: ''.join(map(str, t_args)) if isinstance(t_args, tuple) else ''
                # result = chat_with_qianfan(concatenate_if_tuple(args), tool_input)

                # "你是一个助手，目标是完成用户需求，当前用户需求是：分析给定的本周任务和岗位，根据岗位要求和进度，编写周报。"
                # "岗位和进度:{job_requirements}\n'job_requirements': '工作岗位是数据标注员，本周标注了1000条数据，标注要求是按照标注规范进行标注，标注完成后需要提交给审核人员审核，目前完成了80%，预计下周完成。'。"
                # "以上是用户需求和目标，但你需要根据用户的每一步要求进行回答，而不是直接根据目标和信息给出结果。",
                result = chat_with_qianfan(
                    "你是一个助手，目标是完成用户需求，你需要根据用户的每一步要求进行回答，而不是直接根据目标和信息给出结果。",
                    tool_input)
                print('Qianfan API响应:', result)
                return result
            except Exception as e:
                print(f"An error occurred while calling Qianfan API: {str(e)}")
                return "调用Qianfan API时出错"
        else:
            response = requests.post(self.api_url, headers=headers, json=data)
            result = response.json()
            return result.get("response", "CustomAI没有响应")

    async def _arun(self, tool_input: str, *args) -> str:
        """运行方法的异步版本"""
        raise NotImplementedError("CustomAITool不支持异步")

  4. 执行脚本crew.py

    import asyncio
from dotenv import load_dotenv

# 加载 .env 文件
load_dotenv()

from crewai import Agent, Task, Crew, Process
from tool.custom_tool import CustomAITool


# 初始化自定义工具
custom_ai_tool = CustomAITool()

# 定义代理
researcher = Agent(
    role='中文周报编写人员',
    goal='根据工作岗位编写该工作的周报',
    verbose=True,
    memory=True,
    backstory=(
        "你擅长把日常任务进行周总结。"
        "你的编写风格清晰、简洁"
        "你的编写语言是中文。"
    ),
    tools=[custom_ai_tool]  # 确保工具实例正确传递
)

write_task = Task(
    description=(
        "分析给定的本周任务和岗位，根据岗位要求和进度，编写周报。岗位和进度:{job_requirements}"
    ),
    expected_output='一份汇总详细报告，以markdown格式编写',
    tools=[custom_ai_tool],  # 确保工具实例正确传递
    agent=researcher,
    async_execution=False,
)

# 组建团队
crew = Crew(
    agents=[researcher],
    tasks=[write_task],
    process=Process.sequential
)


def startCrew():
    result = crew.kickoff(inputs={
        'job_requirements': '工作岗位是数据标注员，本周标注了1000条数据，标注要求是按照标注规范进行标注，标注完成后需要提交给审核人员审核，目前完成了80%，预计下周完成。'
    })
    print(f'Kickoff执行结果: {result}')


startCrew()
